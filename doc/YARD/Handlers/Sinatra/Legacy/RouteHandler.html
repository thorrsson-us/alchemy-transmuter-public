def http_path(include_prefix=true)
  path = statement.tokens.find {|t| t.class == YARD::Parser::Ruby::Legacy::RubyToken::TkSTRING }
  path = path ? path.text : ''
  path = $1 if path =~ /^["'](.*)["']/
  include_prefix ? AbstractRouteHandler.uri_prefix + path : path
end
